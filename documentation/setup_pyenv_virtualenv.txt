# To uninstall the different python versions along with the environment
pyenv uninstall 3.8.2

# To activate pyenv environment
export PYENV_ROOT="$HOME/.pyenv"
export PATH="$PYENV_ROOT/bin:$PATH"
eval "$(pyenv init --path)"
eval "$(pyenv init -)"
eval "$(pyenv virtualenv-init -)"
pyenv shell my_env

# To deactivate the pyenv environment
pyenv shell system
